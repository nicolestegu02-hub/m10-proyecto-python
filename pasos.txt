1. INICIAR EL REPOSITORIO LOCAL

    git init
 
2. ENLAZAR CON REPOSITORIO REMOTO

    git remote add origin git@github.com:NelsonRamirezC/m10-proyecto-python.git
 
3. VERIFICAR SI QUEDÓ ENLAZADO EL REPOSITORIO REMOTO

    git remote -v
 
4. CAMBIAR A RAMA MAIN 

    git branch -M main
 
5. AGREGAR CAMBIOS DEL PROYECTO AL ÁREA DE PREPARACIÓN STAGE AGREGAR

    git add .
 
6. VERSIONAR DE FORMA LOCAL EL PROYECTO CON LOS ARCHIVOS SELECCIONADOS

    git commit -m "Agregando archivos .csv para iniciar proyecto de python"
 
7. SUBIR LA VERSIÓN LOCAL AL REPOSITORIO REMOTO (EN LA NUBE)

    git push origin main
    
 8. PASOS QUE SE REPITEN
    git add .
    git commit -m "mensaje"
    git push origin main
